// This is main function
local function printTable(tbl)
    for x = 1, #tbl do
        local str = ""
        for y = 1, #tbl[x] do
            str = str .. tbl[x][y] .. " "
        end
        print(str, "\n")
    end
end

local function printTableBool(tbl)
    for x = 1, #tbl do
        local str = ""
        for y = 1, #tbl[x] do
            if tbl[x][y] then
                str = str .. "T "
            else
                str = str .. "F "
            end
        end
        print(str, "\n")
    end
end


// Example 1
local tableSize = 10
local rndTable = {}
for x = 1, tableSize do
    table.insert(rndTable, {})
    for y = 1, tableSize do
        table.insert(rndTable[x], math.random(0, 9))
    end
end

printTable(rndTable)

local tableSize = 9
