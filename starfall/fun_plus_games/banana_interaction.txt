chipEntId = chip():entIndex()

currentCheckId = 0

hook.add("tick", "SetHealthToEntity", function()
    while (true) do
        if currentCheckId >= chipEntId then
            hook.remove("tick", "SetHealthToEntity")
            chip():remove()
        end
        
        if cpuTotalAverage() > cpuMax() * 0.9 then return end
        currentCheckId = currentCheckId + 1
        ent = entity(currentCheckId)
        
        if not isValid(ent) then continue end
        if not isValid(ent:getOwner()) then continue end
        if ent:getOwner() != owner() then continue end
        
        
        if ent:getModel() == "models/props_farm/scenes/bananana_peel.mdl" then
            ent:setColor(Color(0, 0, 0, 0))
            // ent:remove()
        end
    end
end)
