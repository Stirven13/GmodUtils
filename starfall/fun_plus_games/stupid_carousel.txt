local speedInDegree = 90

hook.add("tick", "ChangeAnglesOfCarousel", function()
    local anglesCarousel = chip():getAngles()
    chip():setAngles(anglesCarousel + Angle(0, speedInDegree, 0))
    chip():setNocollideAll(true)
end)
