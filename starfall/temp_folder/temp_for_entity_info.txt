hook.add("postdrawhud", "DrawHudAboutStatusUsingSizeChecker", function()
        
        // ---------------------------
        // |                         |
        // |                         |
        // |                   _____ |
        // |            +     |     ||
        // |                  |_____||
        // |                         |
        // |_________________________|
        
        render.setColor(configCustomHud.outlineColor)
        render.drawRoundedBox(5, calculatedCoordinates.mainBox.x, calculatedCoordinates.mainBox.y, calculatedCoordinates.mainBox.w, calculatedCoordinates.mainBox.h)
        render.setColor(configCustomHud.backgroundColor)
        render.drawRoundedBox(2, calculatedCoordinates.backgroundBox.x, calculatedCoordinates.backgroundBox.y, calculatedCoordinates.backgroundBox.w, calculatedCoordinates.backgroundBox.h)
        
        render.setColor(configCustomHud.textColor)
        render.setFont(render.createFont("Arial", 20))
        render.drawText(calculatedCoordinates.titleText.x, calculatedCoordinates.titleText.y, "Size Checker tool", 1)
        
        render.drawLine(calculatedCoordinates.lineUnderTitle.x1, calculatedCoordinates.lineUnderTitle.y1, 
                        calculatedCoordinates.lineUnderTitle.x2, calculatedCoordinates.lineUnderTitle.y2) 
    end)
    
    



    
    local screenSizeX, screenSizeY = render.getGameResolution()
    local configCustomHud = ConfigCustomHud:new()

    local calculatedCoordinates = {    
        mainBox = {
            x = screenSizeX * (1 - configCustomHud.widthPercent),
            y = screenSizeY / 2,
            w = screenSizeX * configCustomHud.widthPercent - configCustomHud.indentX,
            h = screenSizeY * configCustomHud.heightPercent
        },
        backgroundBox = {
            x = screenSizeX * (1 - configCustomHud.widthPercent) + 3,
            y = screenSizeY / 2 + 3,
            w = screenSizeX * configCustomHud.widthPercent - configCustomHud.indentX - 6,
            h = screenSizeY * configCustomHud.heightPercent- 6
        },
        titleText = {
            x = screenSizeX * (1 - configCustomHud.widthPercent) + (screenSizeY * configCustomHud.heightPercent) / 2,
            y = screenSizeY / 2 + 8
        },
        lineUnderTitle = {
            x1 = screenSizeX * (1 - configCustomHud.widthPercent) + 13,
            y1 = screenSizeY / 2 + 23,
            x2 = screenSizeX - 13,
            y2 = screenSizeY / 2 + 23
        }
        
    }
    
    
    
    
    local ConfigCustomHud = class("ConfigCustomHud")

function ConfigCustomHud:initialize()
    self.indentX = 10
    self.widthPercent = 0.2
    self.heightPercent = 0.2
    
    self.backgroundColor = Color(202, 240, 248)
    self.textColor = Color(0, 119, 182)
    self.outlineColor = Color(3, 4, 94)
end
